```python
# app/services/inventory_alert.py

class InventoryAlertService:
    @staticmethod
    def check_low_stock():
        """检查低库存并生成预警"""
        low_stock_parts = SparePart.query.filter(
            SparePart.current_stock <= SparePart.min_stock
        ).all()
        
        alerts = []
        for part in low_stock_parts:
            alert = {
                'part_id': part.id,
                'part_code': part.part_code,
                'part_name': part.name,
                'current_stock': part.current_stock,
                'min_stock': part.min_stock,
                'alert_level': 'critical' if part.current_stock == 0 else 'warning'
            }
            alerts.append(alert)
        
        return alerts
    
    @staticmethod
    def create_purchase_suggestion(alerts):
        """生成采购建议"""
        suggestions = []
        for alert in alerts:
            part = SparePart.query.get(alert['part_id'])
            suggested_qty = (part.max_stock or part.min_stock * 2) - part.current_stock
            
            suggestion = {
                'part_id': part.id,
                'part_code': part.part_code,
                'part_name': part.name,
                'suggested_quantity': suggested_qty,
                'estimated_cost': suggested_qty * float(part.unit_price or 0),
                'supplier': part.supplier.name if part.supplier else None
            }
            suggestions.append(suggestion)
        
        return suggestions
```

```python
@app.route('/api/spare_parts/reorder_suggestions')
@login_required
def reorder_suggestions():
    """获取智能补货建议"""
    # 基于历史使用数据
    parts = SparePart.query.all()
    suggestions = []
    
    for part in parts:
        # 计算平均每日消耗
        usage_data = Transaction.query.filter_by(
            spare_part_id=part.id,
            transaction_type='out'
        ).all()
        
        if usage_data:
            total_usage = sum(t.quantity for t in usage_data)
            days = (datetime.now() - min(t.created_at for t in usage_data)).days
            daily_avg = total_usage / max(days, 1)
            
            # 计算建议采购量
            if part.current_stock < daily_avg * 30:  # 少于 30 天用量
                suggestions.append({
                    'part': part,
                    'daily_usage': daily_avg,
                    'days_remaining': part.current_stock / max(daily_avg, 0.1),
                    'suggested_qty': daily_avg * 60 - part.current_stock
                })
    
    return jsonify(sorted(suggestions, key=lambda x: x['days_remaining'])[:20])
```

```python
@login_required
def spare_parts_analytics():
    """备件数据分析"""
    # 库存价值分析
    total_value = db.session.query(
        db.func.sum(SparePart.current_stock * SparePart.unit_price)
    ).scalar()
    
    # 分类统计
    category_stats = db.session.query(
        Category.name,
        db.func.count(SparePart.id),
        db.func.sum(SparePart.current_stock)
    ).join(SparePart).group_by(Category.id).all()
    
    # 库存状态分布
    stock_status_dist = db.session.query(
        SparePart.stock_status,
        db.func.count(SparePart.id)
```

### 4. 性能优化建议

#### ⚠️ 当前性能问题

1. **N+1 查询问题**
   - 列表页面每条记录查询分类和供应商
   - 可能导致性能问题

**解决方案**：
```python
# 使用 joinedload 预加载关联数据
from sqlalchemy.orm import joinedload

query = SparePart.query.options(
    joinedload(SparePart.category),
    joinedload(SparePart.supplier)
)
```

2. **大数据量分页**
   - 使用 offset 分页在大数据量时性能差

**解决方案**：
```python
# 使用键集分页（Keyset Pagination）
def keyset_pagination(query, per_page=20, last_id=None):
    if last_id:
        query = query.filter(SparePart.id > last_id)
    
    items = query.order_by(SparePart.id).limit(per_page + 1).all()
    
    has_next = len(items) > per_page
    if has_next:
        items = items[:per_page]
    
    next_cursor = items[-1].id if items and has_next else None
    
    return {
        'items': items,
        'has_next': has_next,
        'next_cursor': next_cursor
    }
```
