# 备件与仓库联动管理系统 - 完整实现方案

## 📋 联动架构设计

### 1. 数据联动架构

```
备件主表 (spare_part)
    │
    │ 1:N (一个备件可以在多个仓库有多个库存记录)
    │
库存记录表 (inventory_record) ← 核心联动表
    │
    ├→ 仓库表 (warehouse)
    ├→ 货位表 (warehouse_location)
    └→ 出入库单表 (inbound/outbound_order)
```

**核心设计原则：**
1. ✅ **备件主表保持不变** - 保留现有 100 个备件数据
2. ✅ **库存记录细化** - 每个仓库/货位的库存独立记录
3. ✅ **自动同步机制** - 库存记录变化自动汇总到备件主表
4. ✅ **双向关联** - 备件可查所有仓库库存，仓库可查所有备件

### 2. 数据库联动设计

#### 2.1 核心联动表 - 库存记录表

```sql
CREATE TABLE inventory_record (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    
    -- 三要素关联（核心）
    warehouse_id BIGINT NOT NULL COMMENT '仓库 ID',
    location_id BIGINT NOT NULL COMMENT '货位 ID',
    spare_part_id BIGINT NOT NULL COMMENT '备件 ID',
    
    -- 批次管理
    batch_number VARCHAR(100) COMMENT '批次号',
    
    -- 数量信息（联动关键）
    quantity INT NOT NULL DEFAULT 0 COMMENT '当前数量',
    initial_quantity INT DEFAULT 0 COMMENT '初始数量',
    locked_quantity INT DEFAULT 0 COMMENT '锁定数量（已分配未出库）',
    available_quantity INT GENERATED ALWAYS AS (quantity - locked_quantity) COMMENT '可用数量',
    
    -- 价值信息
    unit_price DECIMAL(10,2) DEFAULT 0 COMMENT '入库单价',
    total_value DECIMAL(10,2) DEFAULT 0 COMMENT '总价值',
    
    -- 时间信息
    entry_date DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '入库日期',
    expiry_date DATETIME COMMENT '有效期',
    last_check_date DATETIME COMMENT '最后盘点日期',
    
    -- 库龄信息
    stock_age_days INT DEFAULT 0 COMMENT '库存天数',
    
    -- 状态
    status VARCHAR(20) DEFAULT 'normal' COMMENT '状态：normal/locked/damaged/expired',
    
    -- 操作信息
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    created_by BIGINT COMMENT '创建人 ID',
    
    -- 索引（提升查询性能）
    INDEX idx_warehouse (warehouse_id),
    INDEX idx_location (location_id),
    INDEX idx_spare_part (spare_part_id),
    INDEX idx_status (status),
    
    -- 唯一约束（同一批次同一货位只有一条记录）
    UNIQUE KEY uk_warehouse_location_part_batch (warehouse_id, location_id, spare_part_id, batch_number),
    
    -- 外键（建立关联关系）
    FOREIGN KEY (warehouse_id) REFERENCES warehouse(id),
    FOREIGN KEY (location_id) REFERENCES warehouse_location(id),
    FOREIGN KEY (spare_part_id) REFERENCES spare_part(id),
    FOREIGN KEY (created_by) REFERENCES user(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='库存记录表 - 备件与仓库联动核心表';
```

#### 2.2 触发器 - 自动同步库存到备件主表

```sql
DELIMITER $$

-- 创建触发器：库存记录插入时自动更新备件主表
CREATE TRIGGER trg_inventory_insert_after
AFTER INSERT ON inventory_record
FOR EACH ROW
BEGIN
    -- 计算该备件在所有仓库的总库存
    DECLARE total_stock INT;
    
    SELECT COALESCE(SUM(quantity), 0) INTO total_stock
    FROM inventory_record
    WHERE spare_part_id = NEW.spare_part_id;
    
    -- 更新备件主表的库存
    UPDATE spare_part
    SET current_stock = total_stock,
        updated_at = NOW()
    WHERE id = NEW.spare_part_id;
END$$

-- 创建触发器：库存记录更新时自动更新备件主表
CREATE TRIGGER trg_inventory_update_after
AFTER UPDATE ON inventory_record
FOR EACH ROW
BEGIN
    DECLARE total_stock INT;
    
    SELECT COALESCE(SUM(quantity), 0) INTO total_stock
    FROM inventory_record
    WHERE spare_part_id = NEW.spare_part_id;
    
    UPDATE spare_part
    SET current_stock = total_stock,
        updated_at = NOW()
    WHERE id = NEW.spare_part_id;
    
    -- 如果库存状态发生变化，记录日志
    IF OLD.quantity != NEW.quantity THEN
        INSERT INTO inventory_change_log (
            spare_part_id, warehouse_id, location_id,
            old_quantity, new_quantity, change_type,
            created_at
        ) VALUES (
            NEW.spare_part_id, NEW.warehouse_id, NEW.location_id,
            OLD.quantity, NEW.quantity, 'ADJUSTMENT', NOW()
        );
    END IF;
END$$

-- 创建触发器：库存记录删除时自动更新备件主表
CREATE TRIGGER trg_inventory_delete_after
AFTER DELETE ON inventory_record
FOR EACH ROW
BEGIN
    DECLARE total_stock INT;
    
    SELECT COALESCE(SUM(quantity), 0) INTO total_stock
    FROM inventory_record
    WHERE spare_part_id = OLD.spare_part_id;
    
    UPDATE spare_part
    SET current_stock = total_stock,
        updated_at = NOW()
    WHERE id = OLD.spare_part_id;
END$$

DELIMITER ;
```

```python
# app/services/inbound_linkage_service.py

from datetime import datetime
from flask import current_app
from app.extensions import db
from app.models.spare_part import SparePart
from app.models.inventory_record import InventoryRecord
from app.models.inbound_outbound import InboundOrder, InboundOrderItem


class InboundLinkageService:
    """
    入库联动服务
    实现：入库单 → 库存记录 → 备件主表 的自动联动
    """
    
    @staticmethod
    async def execute_inbound(inbound_order_id, operator_id=None):
        """
        执行入库联动
        
        流程：
        1. 确认入库单
        2. 创建/更新库存记录
        3. 自动触发触发器更新备件主表
        4. 记录库存变动日志
        5. 触发 AI 分析（可选）
        """
        try:
            # 1. 获取入库单
            order = InboundOrder.query.get(inbound_order_id)
            if not order:
                raise ValueError("入库单不存在")
            
            # 2. 处理每个入库明细
            for item in order.items:
                await InboundLinkageService._process_inbound_item(
                    order=order,
                    item=item,
                    operator_id=operator_id
                )
            
            # 3. 更新入库单状态
            order.status = 'completed'
            order.completed_at = datetime.utcnow()
            order.completed_by = operator_id
            
            # 4. 记录操作日志
            await OperationLogService.log({
                'order_type': 'inbound',
                'order_id': order.id,
                'action': 'complete',
                'operator_id': operator_id,
                'details': f"入库完成，共{len(order.items)}项"
            })
            
            # 5. 触发 AI 分析（异步）
            if order.ai_recommended:
                await AIAnalysisService.analyze_inbound_result(order)
            
            db.session.commit()
            
            current_app.logger.info(f"入库联动完成：{order.order_number}")
            return True
            
        except Exception as e:
            db.session.rollback()
            current_app.logger.error(f"入库联动失败：{str(e)}")
            raise
    
    @staticmethod
    async def _process_inbound_item(order, item, operator_id):
        """
        处理单个入库明细项
        
        核心逻辑：创建或更新库存记录
        """
        # 查询是否已存在同批次库存记录
        existing_record = InventoryRecord.query.filter_by(
            warehouse_id=order.warehouse_id,
            location_id=item.location_id,
            spare_part_id=item.spare_part_id,
            batch_number=item.batch_number
        ).first()
        
        if existing_record:
            # 更新现有库存记录
            existing_record.quantity += item.actual_quantity
            existing_record.initial_quantity += item.actual_quantity
            existing_record.unit_price = item.unit_price
            existing_record.total_value = existing_record.quantity * item.unit_price
            existing_record.updated_at = datetime.utcnow()
            
            current_app.logger.info(
                f"更新库存记录：{existing_record.id}, "
                f"数量：{existing_record.quantity}"
            )
        else:
            # 创建新库存记录
            new_record = InventoryRecord(
                warehouse_id=order.warehouse_id,
                location_id=item.location_id,
                spare_part_id=item.spare_part_id,
                batch_number=item.batch_number,
                quantity=item.actual_quantity,
                initial_quantity=item.actual_quantity,
                unit_price=item.unit_price,
                total_value=item.actual_quantity * item.unit_price,
                entry_date=datetime.utcnow(),
                production_date=item.production_date,
                expiry_date=item.expiry_date,
                created_by=operator_id
            )
            db.session.add(new_record)
            
            current_app.logger.info(
                f"创建库存记录：备件{item.spare_part_id}, "
                f"仓库{order.warehouse_id}, 货位{item.location_id}"
            )
        
        # 记录库存变动日志
        await InboundLinkageService._log_inventory_change(
            spare_part_id=item.spare_part_id,
            warehouse_id=order.warehouse_id,
            location_id=item.location_id,
            old_quantity=existing_record.quantity - item.actual_quantity if existing_record else 0,
            new_quantity=existing_record.quantity + item.actual_quantity if not existing_record else existing_record.quantity,
            change_type='INBOUND',
            order_type='inbound',
            order_id=order.id,
            order_number=order.order_number,
            operator_id=operator_id
        )
    
    @staticmethod
    async def _log_inventory_change(**kwargs):
        """记录库存变动日志"""
        from app.models.inventory import InventoryChangeLog
        
        log = InventoryChangeLog(**kwargs)
        db.session.add(log)
```

#### 3.2 出库联动流程

```python
# app/services/outbound_linkage_service.py

from app.models.spare_part import SparePart
from app.models.inventory_record import InventoryRecord
from app.models.inbound_outbound import OutboundOrder, OutboundOrderItem, OutboundSourceLocation


class OutboundLinkageService:
    """
    出库联动服务
    实现：出库单 → 扣减库存记录 → 备件主表自动更新
    """
    
    @staticmethod
    async def execute_outbound(outbound_order_id, operator_id=None):
        """
        执行出库联动
        
        流程：
        1. 确认出库单
        2. 根据策略（FIFO/LIFO/FEFO）选择库存记录
        3. 扣减库存记录
        4. 自动触发触发器更新备件主表
        5. 记录库存变动日志
        """
        try:
            order = OutboundOrder.query.get(outbound_order_id)
            if not order:
                raise ValueError("出库单不存在")
            
            # 1. 锁定库存（防止超卖）
            await OutboundLinkageService._lock_inventory(order)
            
            # 2. 扣减库存
            for item in order.items:
                await OutboundLinkageService._deduct_inventory(
                    order=order,
                    item=item,
                    strategy=order.strategy,
                    operator_id=operator_id
                )
            
            # 3. 更新出库单状态
            order.status = 'completed'
            order.completed_at = datetime.utcnow()
            order.completed_by = operator_id
            
            # 4. 解锁库存
            await OutboundLinkageService._unlock_inventory(order)
            
            # 5. 记录操作日志
            await OperationLogService.log({
                'order_type': 'outbound',
                'order_id': order.id,
                'action': 'complete',
                'operator_id': operator_id
            })
            
            db.session.commit()
            
            current_app.logger.info(f"出库联动完成：{order.order_number}")
            return True
            
        except Exception as e:
            db.session.rollback()
            current_app.logger.error(f"出库联动失败：{str(e)}")
            # 解锁库存
            await OutboundLinkageService._unlock_inventory(order)
            raise
    
    @staticmethod
    async def _lock_inventory(order):
        """锁定库存（预占）"""
        for item in order.items:
            # 查找可用库存
            records = await OutboundLinkageService._find_available_stock(
                spare_part_id=item.spare_part_id,
                warehouse_id=order.warehouse_id,
                quantity=item.requested_quantity,
                strategy=order.strategy
            )
            
            # 锁定库存
            for record in records:
                record.locked_quantity += record.allocate_quantity
            
            # 记录锁定来源
            for record in records:
                source = OutboundSourceLocation(
                    order_item_id=item.id,
                    inventory_record_id=record.id,
                    location_id=record.location_id,
                    quantity=record.allocate_quantity
                )
                db.session.add(source)
    
    @staticmethod
    async def _deduct_inventory(order, item, strategy, operator_id):
        """
        扣减库存
        
        策略：
        - FIFO: 先入库的先出
        - LIFO: 后入库的先出
        - FEFO: 先到期的先出
        """
        # 获取已锁定的库存记录
        sources = OutboundSourceLocation.query.filter_by(
            order_item_id=item.id
        ).all()
        
        old_quantity = 0
        new_quantity = 0
        
        for source in sources:
            record = InventoryRecord.query.get(source.inventory_record_id)
            
            # 记录扣减前数量
            old_quantity = record.quantity
            
            # 扣减库存
            record.quantity -= source.quantity
            record.locked_quantity -= source.quantity
            
            # 记录扣减后数量
            new_quantity = record.quantity
            
            # 记录库存变动日志
            await OutboundLinkageService._log_inventory_change(
                spare_part_id=item.spare_part_id,
                warehouse_id=order.warehouse_id,
                location_id=record.location_id,
                old_quantity=old_quantity,
                new_quantity=new_quantity,
                change_quantity=-source.quantity,
                change_type='OUTBOUND',
                order_type='outbound',
                order_id=order.id,
                order_number=order.order_number,
                operator_id=operator_id
            )
    
    @staticmethod
    async def _find_available_stock(spare_part_id, warehouse_id, quantity, strategy):
        """
        根据策略查找可用库存
        
        Returns:
            list: 库存记录列表（已分配数量）
        """
        # 查询可用库存记录
        query = InventoryRecord.query.filter_by(
            spare_part_id=spare_part_id,
            warehouse_id=warehouse_id,
            status='normal'
        )
        
        # 根据策略排序
        if strategy == 'fifo':
            query = query.order_by(InventoryRecord.entry_date.asc())
        elif strategy == 'lifo':
            query = query.order_by(InventoryRecord.entry_date.desc())
        elif strategy == 'fefo':
            query = query.order_by(
                InventoryRecord.expiry_date.asc().nullslast()
            )
        
        records = query.all()
        
        # 分配库存
        result = []
        remaining = quantity
        
        for record in records:
            available = record.quantity - record.locked_quantity
            if available <= 0:
                continue
            
            allocate = min(available, remaining)
            record.allocate_quantity = allocate
            result.append(record)
            
            remaining -= allocate
            if remaining <= 0:
                break
        
        if remaining > 0:
            raise ValueError(f"库存不足，还需要 {remaining} 个")
        
        return result
    
    @staticmethod
    async def _unlock_inventory(order):
        """解锁库存（异常情况下）"""
        for item in order.items:
            sources = OutboundSourceLocation.query.filter_by(
                order_item_id=item.id
            ).all()
            
            for source in sources:
                record = InventoryRecord.query.get(source.inventory_record_id)
                if record:
                    record.locked_quantity -= source.quantity
```

```python
# app/services/transfer_linkage_service.py

class TransferLinkageService:
    """
    库存调拨联动服务
    实现：仓库 A → 仓库 B 的库存转移
    
    联动特点：
    - 调出仓库库存减少
    - 调入仓库库存增加
    - 备件主表总库存不变
    - 记录调拨轨迹
    """
    
    @staticmethod
    async def execute_transfer(transfer_order_id, operator_id=None):
        """
        执行库存调拨
        """
        try:
            # 获取调拨单
            transfer = InventoryTransfer.query.get(transfer_order_id)
            
            # 1. 从调出仓库扣减库存
            await TransferLinkageService._deduct_source_inventory(transfer)
            
            # 2. 向调入仓库增加库存
            await TransferLinkageService._add_target_inventory(transfer)
            
            # 3. 更新调拨单状态
            transfer.status = 'completed'
            transfer.completed_at = datetime.utcnow()
            
            # 4. 记录调拨日志
            await TransferLinkageService._log_transfer(transfer, operator_id)
            
            db.session.commit()
            
            current_app.logger.info(f"调拨完成：{transfer.transfer_number}")
            return True
            
        except Exception as e:
            db.session.rollback()
            current_app.logger.error(f"调拨失败：{str(e)}")
            raise
    
    @staticmethod
    async def _deduct_source_inventory(transfer):
        """扣减调出仓库库存"""
        for item in transfer.items:
            record = InventoryRecord.query.filter_by(
                warehouse_id=transfer.source_warehouse_id,
                location_id=item.source_location_id,
                spare_part_id=item.spare_part_id
            ).first()
            
            if not record or record.quantity < item.quantity:
                raise ValueError(f"调出仓库库存不足")
            
            record.quantity -= item.quantity
            
            # 记录变动日志
            await TransferLinkageService._log_inventory_change(
                spare_part_id=item.spare_part_id,
                warehouse_id=transfer.source_warehouse_id,
                location_id=item.source_location_id,
                old_quantity=record.quantity + item.quantity,
                new_quantity=record.quantity,
                change_quantity=-item.quantity,
                change_type='TRANSFER_OUT',
                order_type='transfer',
                order_id=transfer.id,
                order_number=transfer.transfer_number
            )
    
    @staticmethod
    async def _add_target_inventory(transfer):
        """增加调入仓库库存"""
        for item in transfer.items:
            # 查找或创建调入仓库库存记录
            record = InventoryRecord.query.filter_by(
                warehouse_id=transfer.target_warehouse_id,
                location_id=item.target_location_id,
                spare_part_id=item.spare_part_id
            ).first()
            
            if record:
                old_quantity = record.quantity
                record.quantity += item.quantity
            else:
                old_quantity = 0
                record = InventoryRecord(
                    warehouse_id=transfer.target_warehouse_id,
                    location_id=item.target_location_id,
                    spare_part_id=item.spare_part_id,
                    quantity=item.quantity,
                    unit_price=item.unit_price
                )
```

```python
# app/services/part_inventory_service.py

class PartInventoryService:
    """
    备件库存分布服务
    查询一个备件在所有仓库的分布情况
    """
    
    @staticmethod
    async def get_part_distribution(spare_part_id):
        """
        获取备件在所有仓库的分布
        
        Returns:
            {
                'part_info': {...},  # 备件基本信息
                'total_stock': 1000,  # 总库存
                'warehouses': [  # 各仓库分布
                    {
                        'warehouse_id': 1,
                        'warehouse_name': '北京仓',
                        'total_quantity': 600,
                        'locations': [
                            {
                                'location_id': 101,
                                'location_code': 'A01-01-01',
                                'quantity': 200,
                                'batch_number': 'BATCH001'
                            }
                        ]
                    }
                ],
                'status_analysis': {
                    'normal': 950,
                    'locked': 50,
                    'damaged': 0
                }
            }
        """
        # 获取备件信息
        part = SparePart.query.get(spare_part_id)
        if not part:
            raise ValueError("备件不存在")
        
        # 查询所有库存记录
        records = InventoryRecord.query.filter_by(
            spare_part_id=spare_part_id
        ).all()
        
        # 按仓库分组
        warehouse_dict = {}
        for record in records:
            wh_id = record.warehouse_id
            if wh_id not in warehouse_dict:
                warehouse_dict[wh_id] = {
                    'warehouse_id': wh_id,
                    'warehouse_name': record.warehouse.name,
                    'total_quantity': 0,
                    'locations': []
                }
            
            warehouse_dict[wh_id]['total_quantity'] += record.quantity
            warehouse_dict[wh_id]['locations'].append({
                'location_id': record.location_id,
                'location_code': record.location.code,
                'quantity': record.quantity,
                'batch_number': record.batch_number or '',
                'status': record.status
            })
        
        # 计算总库存
        total_stock = sum(r.quantity for r in records)
        
        # 状态分析
        status_analysis = {}
        for record in records:
            status = record.status
            if status not in status_analysis:
                status_analysis[status] = 0
            status_analysis[status] += record.quantity
        
        return {
            'part_info': {
                'id': part.id,
                'part_code': part.part_code,
                'name': part.name,
                'specification': part.specification,
                'unit': part.unit
            },
            'total_stock': total_stock,
            'warehouses': list(warehouse_dict.values()),
            'status_analysis': status_analysis
        }
    
    @staticmethod
    async def get_warehouse_inventory(warehouse_id, spare_part_id=None):
        """
        获取仓库的库存（可指定备件）
        
        Returns:
            {
                'warehouse_info': {...},
                'total_skus': 150,  # SKU 数量
                'total_quantity': 5000,  # 总数量
                'total_value': 500000,  # 总价值
                'items': [...]  # 库存明细
            }
        """
        # 查询库存记录
        query = InventoryRecord.query.filter_by(
            warehouse_id=warehouse_id
        )
        
        if spare_part_id:
            query = query.filter_by(spare_part_id=spare_part_id)
        
        records = query.all()
        
        # 统计信息
        total_skus = len(set(r.spare_part_id for r in records))
        total_quantity = sum(r.quantity for r in records)
        total_value = sum(r.total_value for r in records)
        
        # 库存明细
        items = []
        for record in records:
            items.append({
                'part_code': record.spare_part.part_code,
                'part_name': record.spare_part.name,
                'location_code': record.location.code,
                'quantity': record.quantity,
                'unit_price': float(record.unit_price),
                'total_value': float(record.total_value),
                'status': record.status,
                'batch_number': record.batch_number
            })
        
        warehouse = Warehouse.query.get(warehouse_id)
        
        return {
            'warehouse_info': {
                'id': warehouse.id,
                'name': warehouse.name,
                'code': warehouse.code
            },
            'total_skus': total_skus,
            'total_quantity': total_quantity,
            'total_value': total_value,
            'items': items
        }
```

```python
# app/services/inventory_alert_service.py

class InventoryAlertService:
    """
    库存联动预警服务
    监控库存状态，自动触发预警
    """
    
    @staticmethod
    async def check_and_alert():
        """
        检查库存并触发预警
        
        预警类型：
        1. 低库存预警
        2. 超储预警
        3. 库龄预警
        4. 效期预警
        """
        alerts = []
        
        # 1. 低库存预警（按仓库维度）
        low_stock_records = await InventoryRecord.query.filter(
            InventoryRecord.quantity < SparePart.min_stock
        ).join(SparePart).all()
        
        for record in low_stock_records:
            alert = {
                'type': 'low_stock',
                'level': 'warning',
                'warehouse_id': record.warehouse_id,
                'warehouse_name': record.warehouse.name,
                'part_id': record.spare_part_id,
                'part_code': record.spare_part.part_code,
                'current_quantity': record.quantity,
                'min_stock': record.spare_part.min_stock,
                'suggestion': f"建议补货 {record.spare_part.min_stock * 2 - record.quantity} 件"
            }
            alerts.append(alert)
        
        # 2. 超储预警
        over_stock_records = await InventoryRecord.query.filter(
            InventoryRecord.quantity > SparePart.max_stock
        ).join(SparePart).all()
        
        for record in over_stock_records:
            alert = {
                'type': 'overstocked',
                'level': 'info',
                'warehouse_id': record.warehouse_id,
                'part_id': record.spare_part_id,
                'part_code': record.spare_part.part_code,
                'current_quantity': record.quantity,
                'max_stock': record.spare_part.max_stock,
                'suggestion': '建议调拨或促销'
            }
            alerts.append(alert)
        
        # 3. 库龄预警（呆滞料）
        old_stock = await InventoryRecord.query.filter(
            InventoryRecord.stock_age_days > 180,
            InventoryRecord.quantity > 0
        ).all()
        
        for record in old_stock:
            alert = {
                'type': 'slow_moving',
                'level': 'warning',
                'warehouse_id': record.warehouse_id,
                'part_id': record.spare_part_id,
                'part_code': record.spare_part.part_code,
                'stock_age_days': record.stock_age_days,
                'suggestion': '建议优先使用或报废处理'
            }
            alerts.append(alert)
        
        # 4. 效期预警
        expiring_soon = await InventoryRecord.query.filter(
            InventoryRecord.expiry_date != None,
            InventoryRecord.expiry_date <= datetime.utcnow() + timedelta(days=30)
        ).all()
        
        for record in expiring_soon:
            days_to_expire = (record.expiry_date - datetime.utcnow()).days
            alert = {
                'type': 'expiry_warning',
                'level': 'critical',
                'warehouse_id': record.warehouse_id,
                'part_id': record.spare_part_id,
                'part_code': record.spare_part.part_code,
                'expiry_date': record.expiry_date,
                'days_to_expire': days_to_expire,
                'suggestion': f"即将过期，请优先使用"
            }
            alerts.append(alert)
        
        # 发送预警
        for alert in alerts:
            await InventoryAlertService.send_alert(alert)
        
        return alerts
    
    @staticmethod
    async def send_alert(alert):
        """发送预警"""
        # 创建系统通知
        notification = SystemNotification(
            title=f"{alert['type']} - {alert['part_code']}",
            content=alert['suggestion'],
            level=alert['level'],
            related_object_type='inventory',
            related_object_id=alert['part_id']
        )
        db.session.add(notification)
        
        # 发送邮件/短信
        await NotificationService.send({
            'type': 'inventory_alert',
            'recipients': ['warehouse_manager'],
            'alert': alert
        })
```
