# tests/factories.py
import factory
from factory.alchemy import SQLAlchemyModelFactory
from faker import Faker
from app.extensions import db
from app.models.spare_part import SparePart
from app.models.warehouse import Warehouse
from app.models.inventory_record import InventoryRecord

fake = Faker('zh_CN')


class WarehouseFactory(SQLAlchemyModelFactory):
    class Meta:
        model = Warehouse
        sqlalchemy_session = db.session

    name = factory.LazyFunction(lambda: f'仓库-{fake.bothify("##???")}')
    code = factory.Sequence(lambda n: f'WH-{n:04d}')
    type = 'general'
    is_active = True


class SparePartFactory(SQLAlchemyModelFactory):
    class Meta:
        model = SparePart
        sqlalchemy_session = db.session

    part_code = factory.Sequence(lambda n: f'SP-TEST-{n:04d}')
    name = factory.LazyFunction(lambda: f'测试备件-{fake.word()}')
    specification = factory.LazyFunction(lambda: f'{fake.random_int(10,100)}×{fake.random_int(10,100)}')
    current_stock = factory.LazyFunction(lambda: fake.random_int(0, 500))
    min_stock = 10
    max_stock = 200
    safety_stock = 15
    unit = '个'
    unit_price = factory.LazyFunction(lambda: round(fake.random.uniform(5.0, 500.0), 2))
    is_active = True
    stock_status = 'normal'


class InventoryRecordFactory(SQLAlchemyModelFactory):
    class Meta:
        model = InventoryRecord
        sqlalchemy_session = db.session

    spare_part = factory.SubFactory(SparePartFactory)
    warehouse = factory.SubFactory(WarehouseFactory)
    quantity = factory.LazyFunction(lambda: fake.random_int(0, 200))
    available_quantity = factory.LazyFunction(lambda: fake.random_int(0, 200))
    stock_status = 'normal'
