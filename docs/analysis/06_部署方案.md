## 三、本地开发部署

### 3.1 后端开发环境

```bash
# 1. 克隆项目（或进入已有目录）
cd d:/Trae/spare_management

python -m venv venv

source venv/bin/activate
```

## 五、Nginx 配置

### 5.1 HTTPS 站点配置（新增 `/api/transfer`、`/api/reports`、`/api/import` 路由）

编辑 `nginx/conf.d/spare.conf`：

```nginx
# ============================================================
# 备件管理系统 - Nginx 站点配置（含新增功能路由）
# ============================================================

upstream flask_backend {
    least_conn;                         # 最少连接负载均衡
    server backend:5000 max_fails=3 fail_timeout=30s;
    keepalive 32;                       # 保持长连接池
}

# ── HTTP 重定向到 HTTPS ──────────────────────────────────────
server {
    listen 80;
    server_name _;
    location /health { return 200 'OK'; add_header Content-Type text/plain; }
    location / { return 301 https://$host$request_uri; }
}

# ── HTTPS 主站点 ─────────────────────────────────────────────
server {
    listen 443 ssl http2;
    server_name spare.yourdomain.com;

    # SSL 证书（Let's Encrypt 或自签）
    ssl_certificate     /etc/nginx/ssl/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/privkey.pem;
    ssl_protocols       TLSv1.2 TLSv1.3;
    ssl_ciphers         ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_session_cache   shared:SSL:10m;
    ssl_session_timeout 10m;
    ssl_stapling        on;
    ssl_stapling_verify on;
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # ── 静态资源（React SPA）──────────────────────────────────
    root /app/static/react;

    location / {
        try_files $uri $uri/ /index.html;
        expires 1d;
        add_header Cache-Control "public, no-transform";
    }

    # React 构建产物（内容hash，强缓存）
    location ~* /assets/ {
        expires 30d;
        add_header Cache-Control "public, immutable";
    }

    # ── 用户上传文件 ──────────────────────────────────────────
    location /uploads/ {
        alias /app/uploads/;
        expires 7d;
        add_header Cache-Control "public, no-transform";
        # 安全：禁止执行脚本
        location ~* \.(php|py|sh|cgi)$ { deny all; }
    }

    # ── API 路由代理 ──────────────────────────────────────────
    # 原有 API
    location ~ ^/(api|auth|warehouse|spare.parts|spare_parts)/ {
        proxy_pass         http://flask_backend;
        proxy_set_header   Host              $host;
        proxy_set_header   X-Real-IP         $remote_addr;
        proxy_set_header   X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto $scheme;
        proxy_connect_timeout 10s;
        proxy_send_timeout    120s;
        proxy_read_timeout    120s;
        proxy_buffering    off;
        limit_req zone=api_limit burst=50 nodelay;
    }

    # 新增：调拨管理 API（文件上传超时更长）
    location /api/transfer/ {
        proxy_pass         http://flask_backend;
        proxy_set_header   Host              $host;
        proxy_set_header   X-Real-IP         $remote_addr;
        proxy_set_header   X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto $scheme;
        proxy_read_timeout 180s;
        limit_req zone=api_limit burst=30 nodelay;
    }

    # 新增：报表导出 API（大文件传输超时更长）
    location /api/reports/export {
        proxy_pass         http://flask_backend;
        proxy_set_header   Host              $host;
        proxy_set_header   X-Real-IP         $remote_addr;
        proxy_set_header   X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto $scheme;
        proxy_read_timeout    300s;
        proxy_send_timeout    300s;
        proxy_buffering    off;          # 流式下载不缓冲
        limit_req zone=api_limit burst=5 nodelay;
    }

    # 新增：批量导入 API（大文件上传）
    location /api/import/ {
        proxy_pass         http://flask_backend;
        proxy_set_header   Host              $host;
        proxy_set_header   X-Real-IP         $remote_addr;
        proxy_set_header   X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto $scheme;
        proxy_read_timeout    300s;
        client_max_body_size  15M;       # 导入文件最大15MB
        limit_req zone=api_limit burst=5 nodelay;
    }

    # ── 登录限速 ────────────────────────────────────────────
    location /auth/login {
        proxy_pass http://flask_backend;
        proxy_set_header Host            $host;
        proxy_set_header X-Real-IP       $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        limit_req zone=login_limit burst=3 nodelay;
    }

    # ── 健康检查 ────────────────────────────────────────────
    location /health {
        proxy_pass http://flask_backend;
        access_log off;
    }

    # ── 禁止访问敏感路径 ────────────────────────────────────
    location ~* \.(env|git|sql|log|bak)$ { deny all; }
    location ~ /\. { deny all; }
}
```

## 六、数据库迁移流程（生产环境）

### 6.1 执行新功能数据库变更

```bash
# !! 执行前必须备份数据库 !!

# Step 1: 备份
docker compose exec mysql mysqldump \
    -u root -p${MYSQL_ROOT_PASSWORD} \
    spare_parts_db | gzip > backup_before_v1.2.0_$(date +%Y%m%d_%H%M%S).sql.gz

# Step 2: 预览迁移 SQL（不执行）
docker compose exec backend flask db upgrade --sql > migration_preview.sql
cat migration_preview.sql   # 人工审查

# Step 3: 执行迁移
docker compose exec backend flask db upgrade

# Step 4: 验证关键表是否创建成功
docker compose exec mysql mysql -u root -p${MYSQL_ROOT_PASSWORD} spare_parts_db -e "
    SHOW TABLES LIKE 'transfer%';
    SHOW TABLES LIKE 'report%';
    SHOW TABLES LIKE 'import%';
    DESCRIBE spare_part LIMIT 0;
"

# Step 5: 初始化内置报表模板
docker compose exec backend flask shell -c "
from app.services.report_service import init_builtin_reports
init_builtin_reports()
print('报表模板初始化完成')
"

# Step 6: 更新现有备件的FIFO序号（参考03_数据库变更方案.md）
docker compose exec mysql mysql -u root -p${MYSQL_ROOT_PASSWORD} spare_parts_db -e "
UPDATE inventory_record ir
JOIN (
    SELECT id,
           ROW_NUMBER() OVER (PARTITION BY spare_part_id, warehouse_id ORDER BY created_at) AS rn
    FROM inventory_record
) ranked ON ir.id = ranked.id
SET ir.fifo_sequence = ranked.rn;
"
```

### 6.2 回滚方案

```bash
# 方案A：Flask-Migrate 回滚（结构回滚）
docker compose exec backend flask db downgrade

# 方案B：数据库备份恢复（完整回滚）
gunzip -c backup_before_v1.2.0_YYYYMMDD_HHMMSS.sql.gz | \
    docker compose exec -T mysql mysql \
    -u root -p${MYSQL_ROOT_PASSWORD} spare_parts_db

# 回滚后重新部署旧版本代码
```

## 七、APScheduler 定时任务配置

新增功能引入了报表定时发送功能，需确认 `app/scheduler.py` 已注册相关任务：

```python
# app/scheduler.py 中添加报表定时任务（如未添加）

from apscheduler.triggers.cron import CronTrigger

def register_report_schedules(scheduler, app):
    """从数据库读取报表定时任务并注册到 APScheduler"""
    with app.app_context():
        from app.models.report import ReportSchedule
        schedules = ReportSchedule.query.filter_by(is_active=True).all()
        for s in schedules:
            scheduler.add_job(
                func=send_scheduled_report,
                trigger=CronTrigger.from_crontab(s.cron_expression),
                args=[s.id],
                id=f'report_schedule_{s.id}',
                replace_existing=True
            )

def send_scheduled_report(schedule_id):
    """执行报表定时发送"""
    from app.services.report_service import ReportService
    from app.models.report import ReportSchedule
    schedule = ReportSchedule.query.get(schedule_id)
    if schedule and schedule.is_active:
        ReportService.send_scheduled_report(schedule)
```

## 九、SSL 证书配置

### 9.1 使用 Let's Encrypt 免费证书

```bash
# 在宿主机安装 certbot
apt-get install certbot

# 申请证书（需要80端口可访问）
certbot certonly --standalone -d spare.yourdomain.com

# 证书存放目录
ls /etc/letsencrypt/live/spare.yourdomain.com/

# 将证书复制到 nginx/ssl 目录
mkdir -p nginx/ssl
cp /etc/letsencrypt/live/spare.yourdomain.com/fullchain.pem nginx/ssl/
cp /etc/letsencrypt/live/spare.yourdomain.com/privkey.pem nginx/ssl/

# 设置自动续期（crontab）
echo "0 3 1 * * certbot renew --quiet && \
    cp /etc/letsencrypt/live/spare.yourdomain.com/*.pem /path/to/nginx/ssl/ && \
    docker compose restart nginx" | crontab -
```

### 9.2 自签证书（仅开发/内网）

```bash
mkdir -p nginx/ssl
openssl req -x509 -nodes -days 3650 -newkey rsa:2048 \
    -keyout nginx/ssl/privkey.pem \
    -out nginx/ssl/fullchain.pem \
    -subj "/C=CN/ST=Shanghai/L=Shanghai/O=Spare Management/CN=localhost"
```

```bash
#!/bin/bash
# scripts/verify_deployment.sh
# 部署后快速验证脚本

BASE_URL=${1:-"http://localhost"}

echo "=== 备件管理系统 v1.2.0 部署验证 ==="

check() {
    local desc=$1
    local url=$2
    local expected=$3
    local status=$(curl -s -o /dev/null -w "%{http_code}" "$url")
    if [ "$status" = "$expected" ]; then
        echo "✓ $desc ($status)"
    else
        echo "✗ $desc - 期望 $expected, 实际 $status"
    fi
}

check "应用健康检查"          "$BASE_URL/health"                         "200"
```
