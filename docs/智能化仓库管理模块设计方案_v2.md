### 1.2 核心业务模块

#### 1.2.1 仓库基础管理
- **仓库信息管理**：仓库档案、图片、容量、类型
- **库区管理**：存储区、拣货区、收货区、发货区
- **货架管理**：货架类型、层数、承重、规格
- **货位管理**：货位编码、状态、容量、优先级

#### 1.2.2 库存管理
- **实时库存监控**：库存数量、状态、位置
- **批次跟踪**：入库批次、保质期、先进先出
- **库存调拨**：仓库间调拨、库位间移库
- **库存盘点**：定期盘点、循环盘点、动态盘点

#### 1.2.3 入库管理
- **采购入库**：采购订单关联、质检、上架
- **退货入库**：退货验收、重新上架
- **调拨入库**：调拨单关联、入库确认
- **AI 智能入库**：自动识别、自动分类、推荐货位

#### 1.2.4 出库管理
- **销售出库**：销售订单关联、拣货、复核、打包
- **领用出库**：领用申请、审批、发放
- **调拨出库**：调拨单关联、出库确认
- **AI 智能出库**：智能拣货路径、波次优化、自动分拣

#### 1.2.5 智能分析
- **需求预测**：基于历史数据的 AI 预测
- **库存优化**：安全库存建议、补货建议
- **路径优化**：拣货路径规划、效率提升
- **异常检测**：异常出入库检测、风险预警

### 3.2 AI 智能预测实现

#### 3.2.1 需求预测算法

```python
class AIInventoryService:
    """AI 库存服务"""
    
    def __init__(self):
        self.qianfan_service = get_qianfan_service()
    
    def predict_future_demand(self, spare_part_id, days=30):
        """
        AI 预测未来需求
        
        Args:
            spare_part_id: 备件 ID
            days: 预测天数
            
        Returns:
            dict: 预测结果
        """
        # 1. 获取历史数据（过去 90 天）
        historical_data = self._get_historical_transactions(
            spare_part_id, 
            days=90
        )
        
        # 2. 构建 AI 请求
        prompt = self._build_demand_prediction_prompt(
            spare_part_id, 
            historical_data, 
            days
        )
        
        # 3. 调用百度千帆 AI
        result = self.qianfan_service.chat(
            messages=[{'role': 'user', 'content': prompt}],
            system_prompt="你是智能仓库需求预测专家，擅长时间序列分析和趋势预测。"
        )
        
        # 4. 解析结果
        if result['success']:
            prediction = self._parse_prediction_result(result['content'])
            return {
                'success': True,
                'prediction': prediction,
                'confidence': self._calculate_confidence(historical_data)
            }
        else:
            return result
    
    def _get_historical_transactions(self, spare_part_id, days=90):
        """获取历史交易数据"""
        from datetime import datetime, timedelta
        from app.models.transaction import Transaction, TransactionDetail
        
        start_date = datetime.utcnow() - timedelta(days=days)
        
        transactions = db.session.query(
            Transaction.id,
            Transaction.transaction_type,
            TransactionDetail.quantity,
            Transaction.created_at
        ).join(
            TransactionDetail, Transaction.id == TransactionDetail.transaction_id
        ).filter(
            Transaction.spare_part_id == spare_part_id,
            Transaction.created_at >= start_date
        ).order_by(Transaction.created_at.desc()).all()
        
        return [
            {
                'type': t.transaction_type,
                'quantity': t.quantity,
                'date': t.created_at.strftime('%Y-%m-%d')
            }
            for t in transactions
        ]
    
    def _build_demand_prediction_prompt(self, spare_part_id, historical_data, days):
        """构建需求预测提示词"""
        spare_part = SparePart.query.get(spare_part_id)
        
        prompt = f"""
请分析备件 {spare_part.part_code} - {spare_part.name} 的历史出入库数据，预测未来{days}天的需求。

历史数据（共{len(historical_data)}条记录）：
{json.dumps(historical_data, ensure_ascii=False, indent=2)}

请提供以下分析结果（JSON 格式）：
1. 日均出库量：数字
```

```python
class AutomatedInboundService:
    """自动化入库服务"""
    
    def process_inbound_order(self, inbound_order_id):

        处理入库订单（全自动流程）

            # 1. 获取备件信息

            recommendation = ai_service.auto_classify_and_recommend(
                part_info={

                    'category': spare_part.category.name if spare_part.category else None,
                    'dimensions': spare_part.technical_params.get('dimensions') if spare_part.technical_params else None,
                    'weight': spare_part.technical_params.get('weight') if spare_part.technical_params else None
                },
                warehouse_id=inbound_order.warehouse_id
            )
            
            # 3. 生成上架任务
            putaway_task = PutawayTask(
                task_no=f"PUT{datetime.utcnow().strftime('%Y%m%d%H%M%S')}",
                spare_part_id=spare_part.id,
                warehouse_id=inbound_order.warehouse_id,
                location_id=recommendation['location_id'],
                quantity=item.quantity,
                status='pending'
            )
            db.session.add(putaway_task)
            
            # 4. 更新库存记录
            inventory_record = InventoryRecord.query.filter_by(
                spare_part_id=spare_part.id,
                warehouse_id=inbound_order.warehouse_id
            ).first()
            
            if not inventory_record:
                inventory_record = InventoryRecord(
                    spare_part_id=spare_part.id,
                    warehouse_id=inbound_order.warehouse_id,
                    location_id=recommendation['location_id'],
                    quantity=0,
                    min_stock=spare_part.min_stock,
                    max_stock=spare_part.max_stock,
                    safety_stock=spare_part.safety_stock
                )
                db.session.add(inventory_record)
            
            # 5. 更新库存数量
            old_quantity = inventory_record.quantity
            inventory_record.quantity += item.quantity
            inventory_record.available_quantity = inventory_record.quantity - inventory_record.locked_quantity
            inventory_record.after_quantity = inventory_record.quantity
            
            # 6. 创建库存交易记录
            transaction = InventoryTransaction(
                transaction_no=f"IN{datetime.utcnow().strftime('%Y%m%d%H%M%S')}",
                transaction_type='inbound',
                spare_part_id=spare_part.id,
                warehouse_id=inbound_order.warehouse_id,
                location_id=recommendation['location_id'],
                quantity=item.quantity,
                before_quantity=old_quantity,
                after_quantity=inventory_record.quantity,
                reference_type='inbound_order',
                reference_id=inbound_order.id
            )
            db.session.add(transaction)
            
            # 7. 更新备件主表库存
            spare_part.current_stock = InventoryRecord.query.filter_by(
                spare_part_id=spare_part.id
            ).with_entities(db.func.sum(InventoryRecord.quantity)).scalar() or 0
            spare_part.update_stock_status()
        
        db.session.commit()
        
        # 8. 触发 AI 库存健康分析
        ai_health_service = AIInventoryHealthService()
        health_report = ai_health_service.analyze_inventory_health(inbound_order.warehouse_id)
        
        return {
            'success': True,
            'inbound_order_id': inbound_order_id,
            'items_processed': len(inbound_order.items),
            'health_report': health_report
        }
```

```python
        picking_list = []
        for item in outbound_order.items:
            inventory_record = InventoryRecord.query.filter_by(

                warehouse_id=outbound_order.warehouse_id
            ).first()
            
            if inventory_record and inventory_record.available_quantity >= item.quantity:
                picking_list.append({
                    'part_id': item.spare_part_id,
                    'quantity': item.quantity,
                    'location_id': inventory_record.location_id,
                    'priority': 'normal'
                })
            else:
                return {
                    'success': False,
                    'error': f'备件 {item.spare_part.part_code} 库存不足'
                }
        
        # 2. AI 优化拣货路径
        ai_service = AIOutboundService()
        path_optimization = ai_service.optimize_picking_path(
            picking_list=picking_list,
            warehouse_id=outbound_order.warehouse_id
        )
        
        # 3. 生成拣货任务
        picking_task = PickingTask(
            task_no=f"PICK{datetime.utcnow().strftime('%Y%m%d%H%M%S')}",
            outbound_order_id=outbound_order_id,
            warehouse_id=outbound_order.warehouse_id,
            picking_list=picking_list,
            optimized_path=path_optimization['optimized_sequence'],
            estimated_distance=path_optimization['total_distance'],
            estimated_time=path_optimization['estimated_time'],
            status='pending'
        )
        db.session.add(picking_task)
        
        # 4. 执行拣货（模拟）
        for item in picking_list:
            inventory_record = InventoryRecord.query.get_by(
                spare_part_id=item['part_id'],
                warehouse_id=outbound_order.warehouse_id
            )
            
            # 锁定库存
            inventory_record.locked_quantity += item['quantity']
            inventory_record.available_quantity -= item['quantity']
            
            # 创建库存交易记录
            transaction = InventoryTransaction(
                transaction_no=f"OUT{datetime.utcnow().strftime('%Y%m%d%H%M%S')}",
                transaction_type='outbound',
                spare_part_id=item['part_id'],
                warehouse_id=outbound_order.warehouse_id,
                location_id=item['location_id'],
                quantity=item['quantity'],
                before_quantity=inventory_record.quantity,
                after_quantity=inventory_record.quantity - item['quantity'],
                reference_type='outbound_order',
                reference_id=outbound_order_id
            )
            db.session.add(transaction)
        
        db.session.commit()
        
        # 5. 更新备件主表库存
        for item in outbound_order.items:
            spare_part = SparePart.query.get(item.spare_part_id)
            spare_part.current_stock = InventoryRecord.query.filter_by(

            'outbound_order_id': outbound_order_id,
            'picking_task_id': picking_task.id,
            'optimized_path': path_optimization,
```

```python
class InventorySyncService:

    @staticmethod
    def sync_spare_part_inventory(spare_part_id):
        """
        同步备件库存（从分仓库汇总到主表）

        """

        # 更新主表
        spare_part.current_stock = total_stock
        spare_part.update_stock_status()
        
        db.session.commit()
        
        return True
    
    @staticmethod
    def sync_warehouse_statistics(warehouse_id):
        """
        同步仓库统计数据
        
        Args:
            warehouse_id: 仓库 ID
        """
        warehouse = Warehouse.query.get(warehouse_id)
        if not warehouse:
            return False
        
        # 统计总库存
        total_inventory = db.session.query(
            db.func.sum(InventoryRecord.quantity)
        ).filter(
            InventoryRecord.warehouse_id == warehouse_id
        ).scalar() or 0
        
        # 统计备件种类数
        total_spare_parts = db.session.query(
            db.func.count(db.distinct(InventoryRecord.spare_part_id))
        ).filter(
            InventoryRecord.warehouse_id == warehouse_id
        ).scalar() or 0
        
        # 计算利用率
        utilization_rate = 0
        if warehouse.capacity:
            utilization_rate = min(100, (total_inventory / warehouse.capacity) * 100)
        
        # 更新仓库统计字段
        warehouse.total_inventory = total_inventory
        warehouse.total_spare_parts = total_spare_parts
        warehouse.utilization_rate = utilization_rate
        
        db.session.commit()
        
        return True
```
