## 🏗️ 系统架构设计

### 1. 数据联动架构

```
备件 (SparePart)          仓库 (Warehouse)
     │                         │
     │ 1:N                     │ 1:N
     │                         │
     └────→ 库存记录 ←─────────┘
          (InventoryRecord)
                │
                │ N:1
                │
          货位 (WarehouseLocation)
```

**核心设计思路：**
1. 保持现有备件模型不变
2. 新增 `InventoryRecord` 表作为备件与仓库的关联表
3. 备件的 `current_stock` = 所有 `InventoryRecord.quantity` 之和
4. 备件的 `warehouse_id` 和 `location_id` 作为默认值保留

"""
库存记录模型 - 备件与仓库联动的核心模型
"""

from datetime import datetime
from app.extensions import db


class InventoryRecord(db.Model):
    """
    库存记录表
    关联备件、仓库、货位，记录每个库位的库存情况
    """
    __tablename__ = 'inventory_record'
    
    id = db.Column(db.Integer, primary_key=True)
    warehouse_id = db.Column(db.Integer, db.ForeignKey('warehouse.id'), nullable=False, index=True)
    location_id = db.Column(db.Integer, db.ForeignKey('warehouse_location.id'), nullable=False, index=True)
    spare_part_id = db.Column(db.Integer, db.ForeignKey('spare_part.id'), nullable=False, index=True)
    
    # 批次信息
    batch_number = db.Column(db.String(100), index=True)
    
    # 数量信息
    quantity = db.Column(db.Integer, default=0, nullable=False)
    initial_quantity = db.Column(db.Integer, default=0)
    locked_quantity = db.Column(db.Integer, default=0)
    
    # 价值信息
    unit_price = db.Column(db.Numeric(10, 2), default=0.0)
    total_value = db.Column(db.Numeric(10, 2), default=0.0)
    
    # 时间信息
    entry_date = db.Column(db.DateTime, default=datetime.utcnow)
    expiry_date = db.Column(db.DateTime)
    last_check_date = db.Column(db.DateTime)
    
    # 库龄信息
    stock_age_days = db.Column(db.Integer, default=0)
    
    # 状态
    status = db.Column(db.String(20), default='normal', index=True)
    
    # 操作信息
    created_at = db.Column(db.DateTime, default=datetime.utcnow, nullable=False)
    updated_at = db.Column(db.DateTime, default=datetime.utcnow, onupdate=datetime.utcnow, nullable=False)
    created_by = db.Column(db.Integer, db.ForeignKey('user.id'))
    
    # 关联关系
    warehouse = db.relationship('Warehouse', backref=db.backref('inventory_records', lazy='dynamic', cascade='all, delete-orphan'))
    location = db.relationship('WarehouseLocation', backref=db.backref('inventory_records', lazy='dynamic', cascade='all, delete-orphan'))
    spare_part = db.relationship('SparePart', backref=db.backref('inventory_records', lazy='dynamic'))
    creator = db.relationship('User', foreign_keys=[created_by])
    
    # 唯一约束
    __table_args__ = (
        db.UniqueConstraint('warehouse_id', 'location_id', 'spare_part_id', 'batch_number', name='uk_warehouse_location_part_batch'),
    )
    
    def __repr__(self):
        return f'<InventoryRecord {self.id} - Part:{self.spare_part_id} Warehouse:{self.warehouse_id}>'
    
    def to_dict(self):
        """转换为字典"""
        return {
            'id': self.id,
            'warehouse_id': self.warehouse_id,
            'location_id': self.location_id,
            'spare_part_id': self.spare_part_id,
            'batch_number': self.batch_number,
            'quantity': self.quantity,
            'initial_quantity': self.initial_quantity,
            'locked_quantity': self.locked_quantity,
            'unit_price': float(self.unit_price) if self.unit_price else 0,
            'total_value': float(self.total_value) if self.total_value else 0,
            'entry_date': self.entry_date.strftime('%Y-%m-%d') if self.entry_date else None,
            'expiry_date': self.expiry_date.strftime('%Y-%m-%d') if self.expiry_date else None,
            'stock_age_days': self.stock_age_days,
            'status': self.status,
            'created_at': self.created_at.strftime('%Y-%m-%d %H:%M:%S') if self.created_at else None,
            'updated_at': self.updated_at.strftime('%Y-%m-%d %H:%M:%S') if self.updated_at else None,
        }
    
    def is_available(self):
        """是否可用"""
        return self.status == 'normal' and self.quantity > self.locked_quantity
    
    def available_quantity(self):
        """可用数量"""
        return self.quantity - self.locked_quantity
    
    def update_stock_age(self):
        """更新库龄"""
        if self.entry_date:
            self.stock_age_days = (datetime.utcnow() - self.entry_date).days
    
    @classmethod
    def get_total_stock(cls, spare_part_id, warehouse_id=None):
        """
        计算备件的总库存
        
        Args:
            spare_part_id: 备件 ID
            warehouse_id: 仓库 ID（可选，不传则统计所有仓库）
        
        Returns:
            总库存数量
        """
        query = cls.query.filter_by(spare_part_id=spare_part_id)
        if warehouse_id:
            query = query.filter_by(warehouse_id=warehouse_id)
        
        records = query.all()
        return sum(record.quantity for record in records)
    
    @classmethod
    def get_available_stock(cls, spare_part_id, warehouse_id=None):
        """
        计算备件的可用库存
        
        Args:
            spare_part_id: 备件 ID
            warehouse_id: 仓库 ID（可选）
        
        Returns:
            可用库存数量
        """
        query = cls.query.filter_by(spare_part_id=spare_part_id, status='normal')
        if warehouse_id:
            query = query.filter_by(warehouse_id=warehouse_id)
        
        records = query.all()
        return sum(record.available_quantity() for record in records)

                        batch_number=item_data.get('batch_number'),

                    )
                    db.session.add(item)
            
            db.session.commit()
            current_app.logger.info(f"出库单创建成功：{order_number}")
            return order
            
        except Exception as e:
            db.session.rollback()
            current_app.logger.error(f"创建出库单失败：{str(e)}")
            raise
    
    @staticmethod
    def execute_order(order_id, operator_id=None):
        """
        执行出库
        
        Args:
            order_id: 出库单 ID
            operator_id: 操作人 ID
        
        Returns:
            bool: 是否成功
        """
        try:
            order = OutboundOrder.query.get(order_id)
            if not order:
                raise ValueError("出库单不存在")
            
            if order.status not in ['approved', 'pending']:
                raise ValueError("出库单状态不正确")
            
            # 根据策略获取出库库位
            for item in order.items:
                source_locations = InventoryRecordService.get_stock_for_outbound(
                    spare_part_id=item.spare_part_id,
                    warehouse_id=order.warehouse_id,
                    quantity=item.approved_quantity or item.requested_quantity,
                    strategy=order.strategy
                )
                
                # 扣减库存
                for source in source_locations:
                    record = InventoryRecord.query.get(source['inventory_record_id'])
                    record.quantity -= source['quantity']
                    record.locked_quantity -= source['quantity']
                    
                    # 记录出库来源
                    out_source = OutboundSourceLocation(
                        order_item_id=item.id,
                        inventory_record_id=record.id,
                        location_id=record.location_id,
                        quantity=source['quantity']
                    )
                    db.session.add(out_source)
                    
                    item.issued_quantity += source['quantity']
                
                # 更新备件主表库存
                SparePartService.update_stock(item.spare_part_id)
            
            # 更新出库单状态
            order.status = 'completed'
            order.completed_by = operator_id
            order.completed_at = datetime.utcnow()
            
            # 记录操作日志
            OperationLogService.log_operation(
                order_type='outbound',
                order_id=order.id,
                action='execute',
                operator_id=operator_id
            )
            
            db.session.commit()
            current_app.logger.info(f"出库单执行成功：{order.order_number}")
            return True
            
        except Exception as e:
            db.session.rollback()
            current_app.logger.error(f"执行出库失败：{str(e)}")
            raise

"""

from datetime import datetime
from flask import current_app
from app.extensions import db
from app.models.inventory_record import InventoryRecord
from sqlalchemy import and_


class InventoryRecordService:
    """库存记录服务类"""
    
    @staticmethod
    def upsert_inventory(warehouse_id, location_id, spare_part_id, quantity, **kwargs):
        """
        创建或更新库存记录
        
        Args:
            warehouse_id: 仓库 ID
            location_id: 货位 ID
            spare_part_id: 备件 ID
            quantity: 数量
            kwargs: 其他参数
        
        Returns:
            InventoryRecord: 库存记录对象
        """
        try:
            batch_number = kwargs.get('batch_number')
            
            # 查询是否存在
            record = InventoryRecord.query.filter_by(
                warehouse_id=warehouse_id,
                location_id=location_id,
                spare_part_id=spare_part_id,
                batch_number=batch_number
            ).first()
            
            if record:
                # 更新现有记录
                record.quantity += quantity
                record.initial_quantity += quantity
                record.unit_price = kwargs.get('unit_price', record.unit_price)
                record.total_value = record.quantity * record.unit_price
            else:
                # 创建新记录
                record = InventoryRecord(
                    warehouse_id=warehouse_id,
                    location_id=location_id,
                    spare_part_id=spare_part_id,
                    batch_number=batch_number,
                    quantity=quantity,
                    initial_quantity=quantity,
                    unit_price=kwargs.get('unit_price', 0),
                    total_value=quantity * kwargs.get('unit_price', 0),
                    entry_date=datetime.utcnow(),
                    production_date=kwargs.get('production_date'),
                    expiry_date=kwargs.get('expiry_date'),
                    created_by=kwargs.get('operator_id')
                )
                db.session.add(record)
            
            current_app.logger.info(f"库存记录更新成功：{spare_part_id}")
            return record
            
        except Exception as e:
            current_app.logger.error(f"更新库存记录失败：{str(e)}")
            raise
    
    @staticmethod
    def get_stock_for_outbound(spare_part_id, warehouse_id, quantity, strategy='fifo'):
        """
        根据出库策略获取出库库位和数量
        
        Args:
            spare_part_id: 备件 ID
            warehouse_id: 仓库 ID
            quantity: 需要出库的数量
            strategy: 出库策略 (fifo/lifo/fefo)
        
        Returns:
            list: [{inventory_record_id, location_id, quantity}]
        """
        # 查询可用库存
        query = InventoryRecord.query.filter_by(
            spare_part_id=spare_part_id,
            warehouse_id=warehouse_id,
            status='normal'
        )
        
        # 根据策略排序
        if strategy == 'fifo':
            query = query.order_by(InventoryRecord.entry_date.asc())
        elif strategy == 'lifo':
            query = query.order_by(InventoryRecord.entry_date.desc())
        elif strategy == 'fefo':
            query = query.order_by(InventoryRecord.expiry_date.asc().nullslast())
        
        records = query.all()
        
        # 分配出库
        result = []
        remaining = quantity
        
        for record in records:
            available = record.available_quantity()
            if available <= 0:
                continue
            
            take = min(available, remaining)
            result.append({
                'inventory_record_id': record.id,
                'location_id': record.location_id,
                'quantity': take
            })
            
            remaining -= take
            if remaining <= 0:
                break
        
        if remaining > 0:
            raise ValueError(f"库存不足，还需要 {remaining} 个")
        
        return result
    
    @staticmethod
    def get_inventory_summary(warehouse_id=None):
        """
        获取库存汇总
        
        Args:
            warehouse_id: 仓库 ID（可选）
        
        Returns:
            dict: 库存汇总数据
        """
        query = InventoryRecord.query
        
        if warehouse_id:
            query = query.filter_by(warehouse_id=warehouse_id)
        
        records = query.all()
        
        total_quantity = sum(r.quantity for r in records)
        total_value = sum(r.total_value for r in records)
        total_skus = len(set(r.spare_part_id for r in records))
        
        return {
            'total_quantity': total_quantity,
            'total_value': total_value,
            'total_skus': total_skus,
            'warehouse_count': len(set(r.warehouse_id for r in records)),
            'location_count': len(set(r.location_id for r in records))
        }

            order_type=data.get('type', 'requisition'),
            items=data.get('items'),
            strategy=data.get('strategy', 'fifo'),
            operator_id=current_user.id
        )
        
        return jsonify({
            'success': True,
            'message': '出库单创建成功',
            'data': {'order_id': order.id, 'order_number': order.order_number}
        })
        
    except Exception as e:
        current_app.logger.error(f"创建出库单失败：{str(e)}")
        return jsonify({'success': False, 'error': str(e)}), 500

"""
库存记录路由

from flask import Blueprint, request, jsonify, current_app
from flask_login import login_required
from app.models.inventory_record import InventoryRecord
from app.services.inventory_record_service import InventoryRecordService

inventory_record_bp = Blueprint('inventory_record', __name__, url_prefix='/api/inventory-records')


@inventory_record_bp.route('', methods=['GET'])
@login_required
def get_inventory_records():
    """获取库存记录列表"""
    try:
        page = request.args.get('page', 1, type=int)
        per_page = request.args.get('per_page', 50, type=int)
        warehouse_id = request.args.get('warehouse_id', type=int)
        spare_part_id = request.args.get('spare_part_id', type=int)
        status = request.args.get('status')
        
        query = InventoryRecord.query
        
        if warehouse_id:
            query = query.filter_by(warehouse_id=warehouse_id)
        if spare_part_id:
            query = query.filter_by(spare_part_id=spare_part_id)
        if status:
            query = query.filter_by(status=status)
        
        query = query.order_by(InventoryRecord.updated_at.desc())
        pagination = query.paginate(page=page, per_page=per_page, error_out=False)
        
        records = []
        for record in pagination.items:
            record_dict = record.to_dict()
            record_dict['warehouse_name'] = record.warehouse.name if record.warehouse else None
            record_dict['location_code'] = record.location.code if record.location else None
            record_dict['part_code'] = record.spare_part.part_code if record.spare_part else None
            record_dict['part_name'] = record.spare_part.name if record.spare_part else None
            records.append(record_dict)
        
        return jsonify({
            'success': True,
            'data': {
                'records': records,
                'total': pagination.total,
                'pages': pagination.pages,
                'current_page': page
            }
        })
        
    except Exception as e:
        current_app.logger.error(f"获取库存记录失败：{str(e)}")
        return jsonify({'success': False, 'error': str(e)}), 500

### 1. 现有备件数据迁移

```python
"""
将现有备件的库存数据迁移到 InventoryRecord 表
"""

from app.models.spare_part import SparePart
from app.models.inventory_record import InventoryRecord
from app.extensions import db


def migrate_existing_inventory():
    """迁移现有备件库存数据"""
    # 查询所有有库存的备件
    parts = SparePart.query.filter(SparePart.current_stock > 0).all()
    
    for part in parts:
        # 为每个备件创建库存记录
        if part.warehouse_id and part.location_id:
            record = InventoryRecord(
                warehouse_id=part.warehouse_id,
                location_id=part.location_id,
                spare_part_id=part.id,
                quantity=part.current_stock,
                initial_quantity=part.current_stock,
                unit_price=part.unit_price or 0,
                total_value=part.current_stock * (part.unit_price or 0),
                status='normal',
                created_by=1  # 系统迁移
            )
            db.session.add(record)
    
    db.session.commit()
    print(f"迁移完成，共迁移 {len(parts)} 条库存记录")
```
