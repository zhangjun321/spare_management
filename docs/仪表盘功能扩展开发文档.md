## 概述

### 项目背景

当前的备件管理系统仪表盘已经提供了基本的统计功能和图表展示，但在数据洞察、AI分析、用户体验等方面仍有较大提升空间。本文档旨在提供10个详细的功能扩展建议，使系统更加智能化、高效化。

### AI服务确认

系统已集成百度千帆AI服务，使用ERNIE-Bot 4.0模型，通过OpenAI SDK方式调用。相关服务位于：
- `app/services/baidu_qianfan_service.py`

### 存在的问题分析

1. **数据洞察能力不足** - 缺少AI驱动的数据分析和智能建议
2. **时间维度分析缺失** - 无法灵活对比不同时间段的数据
3. **报表导出功能不完善** - 缺少多种格式的报表导出
4. **数据钻取能力有限** - 无法从图表深入查看明细数据
5. **个性化程度低** - 每个用户看到的是相同的布局和数据
6. **KPI目标管理缺失** - 无法设置和追踪业务目标达成情况
7. **预警推送机制不完善** - 缺少实时的异常推送
8. **多维分析能力不足** - 缺少交叉分析和多维度筛选
9. **移动端适配待优化** - 在小屏设备上体验不佳
10. **审计追踪功能缺失** - 谁在什么时间查看了什么数据

```python
# app/services/report_export_service.py - 新增文件

import pandas as pd
from reportlab.lib.pagesizes import A4
from reportlab.pdfgen import canvas
import matplotlib.pyplot as plt
from datetime import datetime

class ReportExportService:
    """报表导出服务"""
    
    def export_to_pdf(self, report_config):
        """导出PDF报表"""
        output_path = f"reports/dashboard_{datetime.now().strftime('%Y%m%d%H%M%S')}.pdf"
        
        c = canvas.Canvas(output_path, pagesize=A4)
        
        # 添加标题
        c.setFont("Helvetica-Bold", 18)
        c.drawString(50, 800, report_config.title)
        
        # 添加图表
        for i, chart in enumerate(report_config.charts):
            # 渲染图表
            self._add_chart_to_pdf(c, chart, 50, 750 - i * 180)
        
        c.save()
        return output_path
    
    def export_to_excel(self, data, output_path=None):
        """导出Excel报表"""
        if not output_path:
            output_path = f"reports/dashboard_{datetime.now().strftime('%Y%m%d%H%M%S')}.xlsx"
        
        with pd.ExcelWriter(output_path, engine='openpyxl') as writer:
            # 概览表
            pd.DataFrame(data['overview']).to_excel(writer, sheet_name='概览', index=False)
            
            # 明细数据表
            for sheet_name, sheet_data in data['details'].items():
                pd.DataFrame(sheet_data).to_excel(writer, sheet_name=sheet_name[:31], index=False)
        
        return output_path
    
    def export_chart_to_png(self, chart_config, output_path=None):
        """导出图表为PNG"""
        # 使用matplotlib渲染图表
        fig, ax = plt.subplots(figsize=chart_config.size)
        self._render_chart(ax, chart_config)
        plt.tight_layout()
        
        if not output_path:
            output_path = f"reports/chart_{datetime.now().strftime('%Y%m%d%H%M%S')}.png"
        
        plt.savefig(output_path, dpi=300, bbox_inches='tight')
        plt.close()
        return output_path
    
    def schedule_report_generation(self, schedule_config):
        """定时生成报表"""
        # 使用APScheduler
        from apscheduler.schedulers.background import BackgroundScheduler
        scheduler = BackgroundScheduler()
        
        scheduler.add_job(
            self._generate_scheduled_report,
            'cron',
            args=[schedule_config],
            **schedule_config.cron_expression
        )
        scheduler.start()
        return True
```

```javascript
function DrillDownPanel({ chartId, data, onDrillDown }) {
    const [breadcrumbs, setBreadcrumbs] = useState([{level: '总览', data: data}]);
    const [currentLevel, setCurrentLevel] = useState('总览');
    const [currentData, setCurrentData] = useState(data);
    
    const handleDrillDown = (item) => {
        // 调用API获取下一级数据
        fetch(`/api/drilldown/${chartId}?value=${item.value}&level=${currentLevel}`)
            .then(res => res.json())
            .then(drillDownData => {
                const newLevel = getNextLevel(currentLevel);
                setBreadcrumbs([...breadcrumbs, {level: newLevel, data: drillDownData, label: item.label}]);
                setCurrentLevel(newLevel);
                setCurrentData(drillDownData);
                onDrillDown(drillDownData);
            });
    };
    
    const handleBack = (index) => {
        // 返回上一级
        const crumb = breadcrumbs[index];
        setBreadcrumbs(breadcrumbs.slice(0, index + 1));
        setCurrentLevel(crumb.level);
        setCurrentData(crumb.data);
    };
    
    return (
        <div className="drilldown-panel">
            <Breadcrumb>
                {breadcrumbs.map((crumb, index) => (
                    <Breadcrumb.Item key={index}>
                        {index === breadcrumbs.length - 1 ? (
                            crumb.label || crumb.level
                        ) : (
                            <a onClick={() => handleBack(index)}>{crumb.label || crumb.level}</a>
                        )}
                    </Breadcrumb.Item>
                ))}
            </Breadcrumb>
            
            <div className="drilldown-content">
                {/* 根据数据类型渲染 */}
                {renderCurrentData(currentData, handleDrillDown)}
            </div>
```

```javascript
                isResizable={isEditMode}
                onLayoutChange={handleLayoutChange}
            >
                {config.components.map(component => (
                    <div key={component.id} className="dashboard-card">
                        <ComponentRenderer component={component} />
                        {isEditMode && (
```

```python
# app/services/analytics_service.py - 增强现有文件

class AnalyticsService:
    """多维分析服务"""
    
    def execute_pivot_query(self, pivot_config):
        """执行数据透视查询"""
        # 构建SQL查询
        base_query = db.session.query(Transaction)
        
        # 应用筛选
        for filter_config in pivot_config.filters:
            base_query = self._apply_filter(base_query, filter_config)
        
        # 分组和聚合
        group_bys = [getattr(Transaction, col) for col in pivot_config.rows]
        aggregations = self._build_aggregations(pivot_config.values)
        
        result = base_query.with_entities(*group_bys, *aggregations) \
                           .group_by(*group_bys) \
                           .all()
        
        # 转换为表格格式
        pivot_table = self._to_pivot_table(result, pivot_config)
        
        return pivot_table
    
    def _build_aggregations(self, values_config):
        """构建聚合函数"""
        aggregations = []
        for val in values_config:
            func_map = {
                'sum': func.sum,
                'count': func.count,
                'avg': func.avg,
                'max': func.max,
                'min': func.min
            }
            agg_func = func_map[val.aggregate]
            aggregations.append(agg_func(getattr(Transaction, val.column)).label(val.alias))
        return aggregations
    
    def calculate_comparative_analysis(self, data, period_type):
        """计算同比/环比分析"""
        result = []
        for i, item in enumerate(data):
            current_value = item.value
            
            # 计算同比 (year over year)
            yoy_value = self._get_comparative_value(data, item.date, 'yoy')
            yoy_rate = (current_value - yoy_value) / yoy_value * 100 if yoy_value else None
            
            # 计算环比 (period over period)
            pop_value = self._get_comparative_value(data, item.date, 'pop')
            pop_rate = (current_value - pop_value) / pop_value * 100 if pop_value else None
            
            result.append({
                **item,
                yoy_value: yoy_value,
                yoy_rate: yoy_rate,
                pop_value: pop_value,
                pop_rate: pop_rate
            })
        
        return result
```
